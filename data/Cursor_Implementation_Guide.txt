STEP 3: Create Supabase Database Tables
────────────────────────────────────────

You'll need these 7 tables in Supabase. Follow these steps:

Go to: https://supabase.com → Your Project → SQL Editor

TABLE 1: customers
────────────────────

Copy & paste this SQL:

```sql
CREATE TABLE customers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  deal_uid TEXT UNIQUE NOT NULL,
  customer_name TEXT NOT NULL,
  website TEXT,
  description TEXT,
  phone TEXT,
  email TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_customers_deal_uid ON customers(deal_uid);
CREATE INDEX idx_customers_name ON customers(customer_name);
```

TABLE 2: customer_locations
──────────────────────────

```sql
CREATE TABLE customer_locations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  customer_id UUID NOT NULL REFERENCES customers(id),
  location_name TEXT,
  address TEXT NOT NULL,
  city TEXT,
  state TEXT,
  zip TEXT,
  phone TEXT,
  email TEXT,
  is_primary BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(customer_id, address)
);

CREATE INDEX idx_locations_customer ON customer_locations(customer_id);
```

TABLE 3: customer_contacts
───────────────────────────

```sql
CREATE TABLE customer_contacts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  customer_id UUID NOT NULL REFERENCES customers(id),
  name TEXT NOT NULL,
  email TEXT,
  phone TEXT,
  role TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_contacts_customer ON customer_contacts(customer_id);
```

TABLE 4: deals
──────────────

```sql
CREATE TABLE deals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  deal_uid TEXT UNIQUE NOT NULL,
  customer_id UUID NOT NULL REFERENCES customers(id),
  provider TEXT NOT NULL,
  product TEXT,
  service_description TEXT,
  monthly_cost NUMERIC(10, 2),
  annual_cost NUMERIC(10, 2),
  users_seats_lines INTEGER,
  contract_start_date DATE,
  contract_end_date DATE,
  renewal_notice_date DATE,
  provider_account_number TEXT,
  source TEXT,
  status TEXT DEFAULT 'ACTIVE',
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_deals_customer ON deals(customer_id);
CREATE INDEX idx_deals_provider ON deals(provider);
CREATE INDEX idx_deals_uid ON deals(deal_uid);
```

TABLE 5: customer_records (for files/documents)
───────────────────────────────────────────────

```sql
CREATE TABLE customer_records (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  deal_id UUID REFERENCES deals(id),
  customer_id UUID REFERENCES customers(id),
  record_type TEXT NOT NULL,
  document_type TEXT,
  file_name TEXT,
  file_path TEXT,
  file_url TEXT,
  location_id UUID REFERENCES customer_locations(id),
  uploaded_date DATE,
  description TEXT,
  visible_in_portal BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_records_customer ON customer_records(customer_id);
CREATE INDEX idx_records_deal ON customer_records(deal_id);
CREATE INDEX idx_records_type ON customer_records(record_type);
```

TABLE 6: partners
─────────────────

```sql
CREATE TABLE partners (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  partner_name TEXT UNIQUE NOT NULL,
  partner_type TEXT NOT NULL,
  description TEXT,
  website TEXT,
  contact_name TEXT,
  contact_email TEXT,
  contact_phone TEXT,
  support_url TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_partners_type ON partners(partner_type);
CREATE INDEX idx_partners_name ON partners(partner_name);
```

TABLE 7: partner_services
───────────────────────────

```sql
CREATE TABLE partner_services (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  partner_id UUID NOT NULL REFERENCES partners(id),
  service_name TEXT NOT NULL,
  description TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_services_partner ON partner_services(partner_id);
```

STEP 7: Create Supabase Integration in Cursor
──────────────────────────────────────────────

In your Cursor project, create a new file:

File: lib/supabase.js (or .ts if TypeScript)

```javascript
import { createClient } from '@supabase/supabase-js'

const supabaseUrl = process.env.REACT_APP_SUPABASE_URL
const supabaseKey = process.env.REACT_APP_SUPABASE_ANON_KEY

export const supabase = createClient(supabaseUrl, supabaseKey)

// Helper function to upload file to Supabase Storage
export async function uploadFile(file, customerDealUid, fileName) {
  const filePath = `customers/${customerDealUid}/${fileName}`
  
  const { data, error } = await supabase.storage
    .from('candid_documents')
    .upload(filePath, file, { upsert: true })
  
  if (error) {
    console.error('Upload error:', error)
    return null
  }
  
  // Get public URL
  const { data: { publicUrl } } = supabase.storage
    .from('candid_documents')
    .getPublicUrl(filePath)
  
  return publicUrl
}

// Helper function to get file URL
export function getFileUrl(customerDealUid, fileName) {
  return supabase.storage
    .from('candid_documents')
    .getPublicUrl(`customers/${customerDealUid}/${fileName}`).data.publicUrl
}
```

STEP 8: Create Import Service Function
────────────────────────────────────────

File: lib/importService.js (or .ts)

```javascript
import { supabase, uploadFile } from './supabase'

export async function importCustomerData(customerJson) {
  const {
    customer,
    locations,
    contacts,
    deals,
    documents
  } = customerJson

  try {
    // 1. Insert Customer
    const { data: customerData, error: customerError } = await supabase
      .from('customers')
      .upsert([
        {
          deal_uid: customer.deal_uid,
          customer_name: customer.customer_name,
          website: customer.website,
          description: customer.description,
          phone: customer.phone,
          email: customer.email
        }
      ], { onConflict: 'deal_uid' })
      .select()
    
    if (customerError) throw customerError
    const customerId = customerData[0].id

    // 2. Insert Locations
    if (locations && locations.length > 0) {
      const locationsData = locations.map(loc => ({
        customer_id: customerId,
        location_name: loc.location_name,
        address: loc.address,
        city: loc.city,
        state: loc.state,
        zip: loc.zip,
        phone: loc.phone,
        email: loc.email,
        is_primary: loc.is_primary || false
      }))

      const { error: locError } = await supabase
        .from('customer_locations')
        .upsert(locationsData, { onConflict: 'customer_id,address' })
      
      if (locError) throw locError
    }

    // 3. Insert Contacts
    if (contacts && contacts.length > 0) {
      const contactsData = contacts.map(con => ({
        customer_id: customerId,
        name: con.name,
        email: con.email,
        phone: con.phone,
        role: con.role
      }))

      const { error: conError } = await supabase
        .from('customer_contacts')
        .insert(contactsData)
      
      if (conError) throw conError
    }

    // 4. Insert Deals
    const dealIds = {}
    if (deals && deals.length > 0) {
      const dealsData = deals.map(deal => ({
        deal_uid: deal.deal_uid,
        customer_id: customerId,
        provider: deal.provider,
        product: deal.product,
        service_description: deal.service_description,
        monthly_cost: deal.monthly_cost,
        annual_cost: deal.annual_cost,
        users_seats_lines: deal.users_seats_lines,
        contract_start_date: deal.contract_start_date,
        contract_end_date: deal.contract_end_date,
        renewal_notice_date: deal.renewal_notice_date,
        provider_account_number: deal.provider_account_number,
        source: deal.source,
        status: deal.status || 'ACTIVE'
      }))

      const { data: dealsInserted, error: dealError } = await supabase
        .from('deals')
        .upsert(dealsData, { onConflict: 'deal_uid' })
        .select()
      
      if (dealError) throw dealError

      dealsInserted.forEach(deal => {
        dealIds[deal.deal_uid] = deal.id
      })
    }

    // 5. Insert Documents/Records
    if (documents && documents.length > 0) {
      const recordsData = documents.map(doc => ({
        deal_id: dealIds[doc.deal_uid],
        customer_id: customerId,
        record_type: doc.record_type,
        document_type: doc.document_type,
        file_name: doc.file_name,
        file_path: `customers/${customer.deal_uid}/${doc.file_name}`,
        uploaded_date: doc.uploaded_date,
        description: doc.description,
        visible_in_portal: true
      }))

      const { error: recError } = await supabase
        .from('customer_records')
        .insert(recordsData)
      
      if (recError) throw recError
    }

    return {
      success: true,
      customerId,
      message: `Imported ${customer.customer_name}`
    }

  } catch (error) {
    console.error('Import error:', error)
    return {
      success: false,
      error: error.message
    }
  }
}

export async function importPartnersData(partnersJson) {
  const { commission_partners, suppliers } = partnersJson
  const allPartners = [...commission_partners, ...suppliers]

  try {
    for (const partner of allPartners) {
      // Insert Partner
      const { data: partnerData, error: partnerError } = await supabase
        .from('partners')
        .upsert([
          {
            partner_name: partner.partner_name,
            partner_type: partner.partner_type,
            description: partner.description,
            website: partner.website,
            support_url: partner.support_url
          }
        ], { onConflict: 'partner_name' })
        .select()
      
      if (partnerError) throw partnerError

      const partnerId = partnerData[0].id

      // Insert Services if they exist
      if (partner.services && partner.services.length > 0) {
        const servicesData = partner.services.map(service => ({
          partner_id: partnerId,
          service_name: typeof service === 'string' ? service : service.name,
          description: typeof service === 'string' ? null : service.description
        }))

        const { error: servError } = await supabase
          .from('partner_services')
          .insert(servicesData)
        
        if (servError) throw servError
      }
    }

    return {
      success: true,
      message: `Imported ${allPartners.length} partners`
    }

  } catch (error) {
    console.error('Partner import error:', error)
    return {
      success: false,
      error: error.message
    }
  }
}
```

import React, { useState } from 'react'
import { importCustomerData, importPartnersData } from '../../lib/importService'

export default function ImportPage() {
  const [status, setStatus] = useState('')
  const [loading, setLoading] = useState(false)
  const [results, setResults] = useState([])

  const handleImportCustomers = async () => {
    setLoading(true)
    setStatus('Importing customers...')
    setResults([])

    try {
      // You'll need to load your JSON files here
      // For now, this is a template
      
      const customerFiles = [
        // Load from your /import_data/customer_jsons/ folder
        // For each file, parse and import
      ]

      const importResults = []
      for (const customerJson of customerFiles) {
        const result = await importCustomerData(customerJson)
        importResults.push(result)
      }

      setResults(importResults)
      setStatus(`✅ Imported ${importResults.length} customers`)
    } catch (error) {
      setStatus(`❌ Error: ${error.message}`)
    } finally {
      setLoading(false)
    }
  }

  const handleImportPartners = async () => {
    setLoading(true)
    setStatus('Importing partners...')

    try {
      // Load partners.json and import
      const partnersJson = {} // Load from /import_data/partners.json

      setResults([result])

      setStatus(`❌ Error: ${error.message}`)
    } finally {
      setLoading(false)
    }
  }

  return (
    <div style={{ padding: '20px' }}>
      <h1>Import Data to Candid Portal</h1>
      
      <div style={{ marginBottom: '20px' }}>

            marginRight: '10px',

            color: 'white',
            border: 'none',
            borderRadius: '4px',
            cursor: loading ? 'not-allowed' : 'pointer'
          }}
        >
          {loading ? 'Importing...' : 'Import Partners'}
        </button>
        
        <button 
          onClick={handleImportCustomers}
          disabled={loading}
          style={{
            padding: '10px 20px',
            backgroundColor: '#28a745',
            color: 'white',
            border: 'none',
            borderRadius: '4px',
            cursor: loading ? 'not-allowed' : 'pointer'

        <div style={{
          padding: '10px',
          marginBottom: '20px',

          borderRadius: '4px'
